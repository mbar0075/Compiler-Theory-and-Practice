fun RandomiseBall(maxXWidth:int,maxYWidth:int,xLimit:int)->int{
    __clear #ffffff;
    let xCount:int =0;
    while(xCount<xLimit){
        __clear #ffffff;
        __pixelr __randi __width,__randi __height,maxXWidth,maxYWidth, #ffff00;
        xCount=xCount+1;
        __delay 70;
    }
    return 2;
}
